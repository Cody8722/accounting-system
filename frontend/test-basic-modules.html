        window.testEventBus = async function() {
            console.log('\n=== 測試 events.js ===\n');
            const container = 'events-tests';
            document.getElementById(container).innerHTML = '';

            try {
                // 清除所有監聽器
                EventBus.clear();

                // 測試 1: 註冊監聽器
                let eventReceived = false;
                EventBus.on('test:event1', () => { eventReceived = true; });
                if (EventBus.getListenerCount('test:event1') === 1) {
                    addTest(container, '註冊監聽器', 'pass', '監聽器計數正確');
                } else {
                    addTest(container, '註冊監聽器', 'fail', '監聽器計數錯誤');
                }

                // 測試 2: 發送事件
                EventBus.emit('test:event1');
                if (eventReceived) {
                    addTest(container, '發送事件', 'pass', '事件正確觸發');
                } else {
                    addTest(container, '發送事件', 'fail', '事件未觸發');
                }

                // 測試 3: 多個監聽器
                let count = 0;
                EventBus.on('test:event2', () => count++);
                EventBus.on('test:event2', () => count++);
                EventBus.emit('test:event2');
                if (count === 2) {
                    addTest(container, '多個監聽器', 'pass', '2 個監聽器都被觸發');
                } else {
                    addTest(container, '多個監聽器', 'fail', `只觸發了 ${count} 個`);
                }

                // 測試 4: 事件數據傳遞
                let receivedData = null;
                EventBus.on('test:event3', (data) => { receivedData = data; });
                EventBus.emit('test:event3', { test: 123 });
                if (receivedData && receivedData.test === 123) {
                    addTest(container, '事件數據傳遞', 'pass', '數據正確傳遞');
                } else {
                    addTest(container, '事件數據傳遞', 'fail', '數據傳遞失敗');
                }

                // 測試 5: 移除監聽器
                const handler = () => {};
                EventBus.on('test:event4', handler);
                EventBus.off('test:event4', handler);
                if (EventBus.getListenerCount('test:event4') === 0) {
                    addTest(container, '移除監聽器', 'pass', '監聽器正確移除');
                } else {
                    addTest(container, '移除監聽器', 'fail', '監聽器未移除');
                }

                // 測試 6: EVENTS 常量
                if (EVENTS.RECORD_ADDED && EVENTS.STATS_UPDATED && EVENTS.AUTH_LOGIN_SUCCESS) {
                    const eventCount = Object.keys(EVENTS).length;
                    addTest(container, 'EVENTS 常量', 'pass', `定義了 ${eventCount} 個事件常量`);
                } else {
                    addTest(container, 'EVENTS 常量', 'fail', 'EVENTS 常量不完整');
                }

            } catch (error) {
                addTest(container, '模組載入', 'fail', error.message);
            }
        };
